 window.vimeoPlayerInstance = new Vimeo.Player(
      document.getElementById("vimeoPlayer")
    );
    document.getElementById("boton-cerrar").addEventListener("click", cerrarVideo);
    document.getElementById("boton-cerrar").addEventListener("touchend", cerrarVideo);

    function cerrarVideo(e) {
      e.preventDefault();

      const modalEl = document.getElementById("videoModal");
      const modal = bootstrap.Modal.getOrCreateInstance(modalEl);
      modal.hide();

      // Pausar Vimeo
      if (window.vimeoPlayerInstance) {
        window.vimeoPlayerInstance.pause();
      }
    }

    document.getElementById("boton-cerrar").addEventListener("touchend", function (e) {
      e.preventDefault();

      const modalEl = document.getElementById("videoModal");
      const modal = bootstrap.Modal.getInstance(modalEl);
      modal.hide();
    });

    document.addEventListener("DOMContentLoaded", function () {
      const iframe = document.getElementById("vimeoPlayer");
      const player = new Vimeo.Player(iframe);
      const modal = document.getElementById("videoModal");

      modal.addEventListener("hidden.bs.modal", function () {
        player.pause();

        document.activeElement.blur();

        document.body.focus();
      });
    });

    window.addEventListener('load', () => {
      const modalEl = document.getElementById('videoModal');
      const modal = new bootstrap.Modal(modalEl);
      const loading = document.getElementById('loadingContent');
      const video = document.getElementById('videoContent');

      modal.show();

      setTimeout(() => {
        loading.classList.add('d-none');
        video.classList.remove('d-none');
      }, 3000);

      modalEl.addEventListener('hidden.bs.modal', () => {
        loading.classList.remove('d-none');
        video.classList.add('d-none');
      });
    });

    const text = "CARGANDO VIDEO";
    const loadingTextEl = document.getElementById("loadingText");
    let index = 0;

    function typeEffect() {
      if (index < text.length) {
        loadingTextEl.textContent += text.charAt(index);
        index++;
        setTimeout(typeEffect, 280);
      }
    }
    typeEffect();
